#### 例子

``` rust
// 定义常量用const. 常量是绑定到一个名称且不允许改变和覆盖的值
const AGE: i32 = 25;
// const AGE = 26; 报错，不允许改变
const NUM: f64 = 233.0;
// const NUM:f64 = 211.0; 报错，已经定义过，不允许覆盖


fn main() {
    // 定义变量用let,依据定义时是否使用mut,变量可以表现出可变或者不可变特性
    let x: f64 = 3.14; // 用let定义变量x，x可被覆盖但不允许改变。
    // x = 6.28; 报错，x不可变
    let x: f64 = 2.71; // 变量下被覆盖

    let mut y = 985; // 用let mut定义变量y，y可被覆盖，并且允许改变。
    y = 996; // 改变y
    let y = 2019; // 变量y被覆盖
    // y = 90; 错误，覆盖后的变量非mut,不能改变

    // 布尔类型用bool表示，只有两个值 true、 false

    // 字符类型char和C语言中的char是一样的，都是最原始的类型。字符用单引号声明，字符串用双引号声明。字符是一个4字节的Unicode标量值
    // 而字符串对应的是数组
    // Unicode标量值，
    let c = 's';
    // 动态数组
    let s_str = "s";

    // 元组是一种能够将多个其他类型的值组成复合值的类型，一旦声明，长度就不能增大或缩小。
    // 为了从元组中获取值可以使用模式匹配和点符号，下标从0开始
    let tup1: (i8, f32, i64) = (-1, 2.33, 8000_0000); // let就是模式匹配，定义变量时也是模式匹配
    let (x, y, z) = tup1; // 使用模式匹配获取所有值
    let tup2 = (0, 100, 2.4);
    let zero = tup2.0; // 使用点符号获取值
    let one_hundred = tup2.1; // 使用点符号获取值


    // 
}

```
